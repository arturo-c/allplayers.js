1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 /**
  5  * @constructor
  6  * @class The base entity class to store the data that is common to all
  7  * drupal entities whether it be groups, events, users, etc.
  8  *
  9  * @param {object} object The entity object.
 10  * @param {function} callback The callback function to get the object.
 11  */
 12 drupal.entity = function(object, callback) {
 13 
 14   // Only continue if the object is valid.
 15   if (object) {
 16 
 17     /** The unique identifier for this entity. */
 18     this.id = this.id || '';
 19 
 20     /** The API for this entity */
 21     this.api = this.api || null;
 22 
 23     // If object is a string, assume it is a UUID and get it.
 24     this.update(object);
 25 
 26     // If they provide a callback, call it now.
 27     if (callback) {
 28 
 29       // Get the object from the server.
 30       this.get(callback);
 31     }
 32   }
 33 };
 34 
 35 /**
 36  * Get's an object from the drupal API.
 37  *
 38  * @param {function} callback The callback function when the object is
 39  * retrieved.
 40  */
 41 drupal.entity.prototype.get = function(callback) {
 42 
 43   // If the API exists, then we need to get the object.
 44   if (this.api) {
 45 
 46     // Call the API.
 47     var _this = this;
 48     this.api.get(this.getObject(), this.getQuery(), function(object) {
 49 
 50       if (!object) {
 51         callback(null);
 52       }
 53       else if (object[0]) {
 54 
 55         var i = object.length;
 56         while (i--) {
 57           object[i] = new _this.constructor(object[i]);
 58         }
 59 
 60         // Callback a list of objects.
 61         callback(object);
 62       }
 63       else {
 64         // Update the object, then call the callback.
 65         _this.update(object);
 66 
 67         if (callback) {
 68           callback(_this);
 69         }
 70       }
 71     });
 72   }
 73 };

 75 /**
 76  * Saves this entity.
 77  *
 78  * @param {function} callback The function called once entity is saved.
 79  */
 80 drupal.entity.prototype.save = function(callback) {
 81 
 82   // If the API exists, then we can save this object.
 83   if (this.api) {
 84 
 85     // Call the API.
 86     var _this = this;
 87     this.api.save(this.getObject(), function(object) {
 88 
 89       // Update the object, then call the callback.
 90       _this.update(object);
 91 
 92       if (callback) {
 93         callback(_this);
 94       }
 95     });
 96   }
 97 };

 99 /**
100  * Removes an entity
101  *
102  * @param {function} callback The function called once entity is removed.
103  */
104 drupal.entity.prototype.remove = function(callback) {
105 
106   // Only remove if they have an ID.
107   if (this.id) {
108 
109     // Call the API.
110     this.api.remove(this.getObject(), callback);
111   }
112 };

114 /**
115  * Returns the search query.
116  *
117  * @return {object} The query to pass to the server.
118  */
119 drupal.entity.prototype.getQuery = function() {
120 
121   var query = {};
122 
123   // We only need to provide a search query if there is no ID.
124   if (!this.id) {
125 
126     // Iterate through all of our fields.
127     for (var field in this) {
128 
129       // Make sure that this property exists, is set, and is not an object.
130       if (this.hasOwnProperty(field) &&
131           this[field] &&
132           (typeof this[field] != 'object')) {
133 
134         // Add this as a search parameter.
135         query['parameters[' + field + ']'] = this[field];
136       }
137     }
138   }
139 
140   // Return the params.
141   return query;
142 };

144 /**
145  * Update the entity data.
146  *
147  * @param {object} object The entity information.
148  */
149 drupal.entity.prototype.update = function(object) {
150 
151   // Update the object.
152   if (object) {
153 
154     // Update the params.
155     for (var param in object) {
156 
157       // Check to make sure that this param is within object scope.
158       if (object.hasOwnProperty(param) && this.hasOwnProperty(param)) {
159 
160         // Check to see if this object has an update function.
161         if (this[param].update) {
162           this[param].update(object[param]);
163         }
164         else {
165           this[param] = object[param];
166         }
167       }
168     }
169   }
170 };
171 
172 /**
173  * Returns the object to send during PUT's and POST's during a save or add.
174  *
175  * @return {object} The JSON object to send to the Services endpoint.
176  */
177 drupal.entity.prototype.getObject = function() {
178   return {
179     id: this.id
180   };
181 };
