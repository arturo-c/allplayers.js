1 /** The allplayers namespace. */
  2 var allplayers = allplayers || {};
  3 
  4 (function($) {
  5 
  6   /** The default options. */
  7   var defaults = {
  8 
  9   };
 10 
 11   // Store all the calendar instances.
 12   allplayers.calendars = {};
 13 
 14   // Add a way to instanciate using jQuery prototype.
 15   if (!$.fn.allplayers_calendar) {
 16     $.fn.allplayers_calendar = function(options) {
 17       return $(this).each(function() {
 18         if (!allplayers.calendars[$(this).selector]) {
 19           new allplayers.calendar($(this), options);
 20         }
 21       });
 22     };
 23   }
 24 
 25   /**
 26    * @class The AllPlayers calendar JavaScript API
 27    *
 28    * <p><strong>Usage:</strong>
 29    * <pre><code>
 30    *
 31    *   // Create a calendar
 32    *   var player = $("#calendar").apcicalendar({
 33    *
 34    *   });
 35    *
 36    * </code></pre>
 37    * </p>
 38    *
 39    * @param {object} context The jQuery context.
 40    * @param {object} options This components options.
 41    */
 42   allplayers.calendar = function(context, options) {
 43     // Make sure we provide default options...
 44     var _this = this;
 45     options = $.extend(defaults, options, {
 46       editable: true,
 47       dayClick: this.onDayClick,
 48       eventClick: this.onEventClick,
 49       eventDragStop: function(event, jsEvent, ui, view) {
 50 
 51         // Save this event.
 52         event.obj.save();
 53       },
 54       events: function(start, end, callback) {
 55         _this.getEvents(start, end, callback);
 56       }
 57     });
 58 
 59     // Store this player instance.
 60     allplayers.calendars[options.id] = this;
 61 
 62     // TO-DO: MAKE IT SO THAT WE DON'T NEED A GROUP TO GET EVENTS
 63     this.uuid = '';
 64 
 65     // The api.
 66     this.api = new allplayers.api();
 67 
 68     // Create the fullcalendar.
 69     context.fullCalendar(options);
 70   };
 71 
 72   allplayers.calendar.prototype.onDayClick = function() {
 73     console.log('Day has been clicked');
 74   };
 75 
 76   allplayers.calendar.prototype.onEventClick = function() {
 77     console.log('Event has been clicked');
 78   };
 79 
 80   allplayers.calendar.prototype.getUUID = function(callback) {
 81     if (this.uuid) {
 82       callback.call(this);
 83     }
 84     else {
 85       var _this = this;
 86       this.api.getGroups('towncenter', function(groups) {
 87         _this.uuid = groups[0].uuid;
 88         callback.call(_this);
 89       });
 90     }
 91   };
 92 
 93   allplayers.calendar.prototype.getEvents = function(start, end, callback) {
 94     var year = end.getFullYear();
 95     var month = end.getMonth();
 96     this.getUUID(function() {
 97       this.api.getGroupEvents(this.uuid, {
 98         month: year + '-' + month,
 99         fields: '*',
100         limit: 0,
101         offset: 0
102       }, function(events) {
103 
104         // Iterate through the events and add a new event object.
105         var i = events.length;
106         while (i--) {
107           events[i].ojb = new allplayers.event(events[i]);
108         }
109 
110         // Add this to the events for the calendar.
111         callback(events);
112       });
113     });
114   };
115 
116 }(jQuery));
