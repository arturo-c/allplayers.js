1 /** The allplayers namespace. */
  2 var allplayers = allplayers || {};
  3 
  4 /**
  5  * @constructor
  6  * @extends drupal.entity
  7  * @class The AllPlayers event class
  8  *
  9  * @param {object} object The node object.
 10  * @param {function} callback The function to be called once the node has
 11  * been retrieved from the server.
 12  * @param {object} options Options used to govern functionality.
 13  */
 14 allplayers.event = function(object, callback, options) {
 15   drupal.node.call(this, object, callback, options);
 16 };
 17 
 18 /** Derive from node. */
 19 allplayers.event.prototype = new drupal.node();
 20 
 21 /** Reset the constructor. */
 22 allplayers.event.prototype.constructor = allplayers.event;
 23 
 24 /** Declare the event api. */
 25 allplayers.event.api = jQuery.extend(new drupal.api(), {
 26   resource: 'events'
 27 });
 28 
 29 /**
 30  * Returns an index of events.
 31  *
 32  * @param {object} query The query parameters.
 33  * @param {function} callback The callback function.
 34  * @param {object} options Options used to govern functionality.
 35  */
 36 allplayers.event.index = function(query, callback, options) {
 37   drupal.entity.index(allplayers.event, query, callback, options);
 38 };
 39 
 40 /**
 41  * Sets the object.
 42  *
 43  * @param {object} object The object which contains the data.
 44  */
 45 allplayers.event.prototype.set = function(object) {
 46   drupal.node.prototype.set.call(this, object);
 47 
 48   /** The name of this entity. */
 49   this.entityName = 'event';
 50 
 51   /** Set the api to the drupal.node.api. */
 52   this.api = allplayers.event.api;
 53 
 54   /** Set the id based on the uuid of the object. */
 55   this.id = this.id || object.uuid || object.id || '';
 56 
 57   // Set the values for this entity.
 58   this.setValues({
 59     allDay: false,
 60     gids: [],
 61     description: '',
 62     resources: [],
 63     competitors: {},
 64     category: 'Other'
 65   }, object);
 66 
 67   /** The date-time object */
 68   this.date = new allplayers.date(object.start, object.end);
 69   this.start = this.date.start;
 70   this.end = this.date.end;
 71 };
 72 
 73 /**
 74  * Returns the object to send to Services.
 75  *
 76  * @return {object} The object to send to the Services endpoint.
 77  */
 78 allplayers.event.prototype.get = function() {
 79   return jQuery.extend(drupal.node.prototype.get.call(this), {
 80     allDay: this.allDay,
 81     gids: this.gids,
 82     description: this.description,
 83     resources: this.resources,
 84     competitors: this.competitors,
 85     category: this.category,
 86     date_time: this.date.get()
 87   });
 88 };
