 40 /**
 41  * Sets the object.
 42  *
 43  * @param {object} object The object which contains the data.
 44  */
 45 allplayers.group.prototype.set = function(object) {
 46   drupal.node.prototype.set.call(this, object);
 47 
 48   /** The name of this entity. */
 49   this.entityName = 'group';
 50 
 51   /** Set the api. */
 52   this.api = allplayers.group.api;
 53 
 54   /** Set the id based on the uuid of the object. */
 55   this.id = object.uuid || object.id || this.id || '';
 56 
 57   /** See if this group has children. */
 58   var has_value = object.hasOwnProperty('has_children');
 59   this.has_children = has_value ? object.has_children : !!this.has_children;
 60 
 61   /** A {@link allplayers.location} object. */
 62   this.location = object.location || this.location || new allplayers.location();
 63 
 64   // Set the values for this entity.
 65   this.setValues({
 66     activity_level: 0,
 67     list_in_directory: 0,
 68     active: false,
 69     registration_fees_enabled: '',
 70     approved_for_payment: '',
 71     accept_amex: '',
 72     primary_color: '',
 73     secondary_color: '',
 74     node_status: 0,
 75     logo: '',
 76     url: '',
 77     groups_above_uuid: [],
 78     registration_link: '',
 79     registration_text: ''
 80   }, object);
 81 };

109 /**
110  * Returns the events for this group.
111  *
112  * @param {object} query An object of the following parameters.
113  * <ul>
114  * <li><strong>start</strong> - The start date to get the events.</li>
115  * <li><strong>end</strong> - The end date to get the events.</li>
116  * <li><strong>fields</strong> - The fields to get.</li>
117  * <li><strong>limit</strong> - The limit of events to get.</li>
118  * <li><strong>offset</strong> - The offset of events for pagination.</li>
119  * </ul>
120  *
121  * @param {function} callback The callback function to get the events.
122  */
123 allplayers.group.prototype.getEvents = function(query, callback) {
124 
125   // Get the events within this group.
126   this.api.get(this, 'events', query, function(events) {
127 
128     // Iterate through the events and create an event object out of them.
129     for (var i in events) {
130       events[i] = new allplayers.event(events[i]);
131     }
132 
133     // Call the callback.
134     callback(events);
135   }, true);
136 };
