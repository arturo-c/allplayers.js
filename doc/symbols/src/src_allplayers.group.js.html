1 /** The allplayers namespace. */
  2 var allplayers = allplayers || {};
  3 
  4 /**
  5  * @constructor
  6  * @extends drupal.entity
  7  * @class The group class to govern all functionality that groups have.
  8  *
  9  * @param {object} object The group information.
 10  * @param {function} callback The function to call when this group is retrieved.
 11  */
 12 allplayers.group = function(object, callback) {
 13 
 14   // Only continue if the object is valid.
 15   if (object) {
 16 
 17     /** A {@link allplayers.location} object. */
 18     this.location = this.location || new allplayers.location();
 19 
 20     /** The group activity level */
 21     this.activity_level = this.activity_level || 0;
 22 
 23     /** List in directory. */
 24     this.list_in_directory = this.list_in_directory || 0;
 25 
 26     /** If the group is active. */
 27     this.active = this.active || false;
 28 
 29     /** Registration fee's enabled */
 30     this.registration_fees_enabled = this.registration_fees_enabled || '';
 31 
 32     /** Approved for payment */
 33     this.approved_for_payment = this.approved_for_payment || '';
 34 
 35     /** Accept AMEX credit cards */
 36     this.accept_amex = this.accept_amex || '';
 37 
 38     /** Primary Color */
 39     this.primary_color = this.primary_color || '';
 40 
 41     /** Secondary Color */
 42     this.secondary_color = this.secondary_color || '';
 43 
 44     /** The node status */
 45     this.node_status = this.node_status || 0;
 46 
 47     /** The group logo URL */
 48     this.logo = this.logo || '';
 49 
 50     /** The Group URI */
 51     this.uri = this.uri || '';
 52 
 53     /** The Group URL */
 54     this.url = this.url || '';
 55 
 56     /** Array of groups above */
 57     this.groups_above_uuid = this.groups_above_uuid || [];
 58 
 59     /** The search parameter for this group. */
 60     this.search = this.search || '';
 61 
 62     /** The groups api. */
 63     this.api = this.api || new allplayers.group.api();
 64   }
 65 
 66   // Derive from drupal.node.
 67   drupal.node.call(this, object, callback);
 68 };

104 /**
105  * Returns the events for this group.
106  *
107  * @param {object} params An object of the following parameters.
108  * <ul>
109  * <li><strong>start</strong> - The start date to get the events.</li>
110  * <li><strong>end</strong> - The end date to get the events.</li>
111  * <li><strong>fields</strong> - The fields to get.</li>
112  * <li><strong>limit</strong> - The limit of events to get.</li>
113  * <li><strong>offset</strong> - The offset of events for pagination.</li>
114  * </ul>
115  *
116  * @param {function} callback The callback function to get the events.
117  */
118 allplayers.group.prototype.getEvents = function(params, callback) {
119 
120   // Get the events within this group.
121   this.api.getItems(this, 'events', params, function(events) {
122 
123     // Iterate through the events and create an event object out of them.
124     var i = events.length;
125     while (i--) {
126       events[i] = new allplayers.event(events[i]);
127     }
128 
129     // Call the callback.
130     callback(events);
131   });
132 };
