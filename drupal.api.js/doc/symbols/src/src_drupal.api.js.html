1 /** The drupal namespace. */
  2 var drupal = drupal || {};
  3 
  4 /**
  5  * @constructor
  6  * @class The base Drupal API class.
  7  */
  8 drupal.api = function() {
  9 
 10   /** The Services API endpoint */
 11   this.endpoint = drupal.endpoint;
 12 
 13   /** The resource within this endpoint */
 14   this.resource = '';
 15 };
 16 
 17 /**
 18  * Helper function to get the Services URL for this resource.
 19  *
 20  * @param {object} object The object involved with in this request.
 21  * @return {string} The path to the API endpoint.
 22  */
 23 drupal.api.prototype.getURL = function(object) {
 24   var path = this.endpoint;
 25   path += this.resource ? ('/' + this.resource) : '';
 26   path += (object && object.id) ? ('/' + object.id) : '';
 27   return path;
 28 };
 29 
 30 /**
 31  * API function to act as a generic request for all Service calls.
 32  *
 33  * @param {string} url The URL where the request will go.
 34  * @param {string} dataType The type of request.  json or jsonp.
 35  * @param {string} type The type of HTTP request.  GET, POST, PUT, etc.
 36  * @param {object} data The data to send to the server.
 37  * @param {function} callback The function callback.
 38  */
 39 drupal.api.prototype.call = function(url, dataType, type, data, callback) {
 40   var request = {
 41     url: url,
 42     dataType: dataType,
 43     type: type,
 44     success: function(data, textStatus) {
 45       if (textStatus == 'success') {
 46         callback(data);
 47       }
 48       else {
 49         console.log('Error: ' + textStatus);
 50       }
 51     },
 52     error: function(xhr, ajaxOptions, thrownError) {
 53       console.log(xhr.responseText);
 54       callback(null);
 55     }
 56   };
 57 
 58   if (data) {
 59     request['data'] = data;
 60   }
 61 
 62   // Make the request.
 63   jQuery.ajax(request);
 64 };
 65 
 66 /**
 67  * API function to get any results from the drupal API.
 68  *
 69  * @param {object} object The object of the item we are getting..
 70  * @param {object} query key-value pairs to add to the query of the URL.
 71  * @param {function} callback The callback function.
 72  */
 73 drupal.api.prototype.get = function(object, query, callback) {
 74   var url = this.getURL(object);
 75   url += '.jsonp';
 76   url += query ? ('?' + decodeURIComponent(jQuery.param(query, true))) : '';
 77   this.call(url, 'jsonp', 'GET', null, callback);
 78 };
 79 
 80 /**
 81  * API function to perform an action.
 82  *
 83  * @param {string} action The action to perform.
 84  * @param {object} object The entity object to set.
 85  * @param {function} callback The callback function.
 86  */
 87 drupal.api.prototype.execute = function(action, object, callback) {
 88   var url = this.getURL(object) + '/' + action;
 89   this.call(url, 'json', 'POST', object, callback);
 90 };
 91 
 92 /**
 93  * API function to save the value of an object using Services.
 94  *
 95  * @param {object} object The entity object to set.  If the object does not
 96  * have an ID, then this will create a new entity, otherwise, it will simply
 97  * update the existing resource.
 98  *
 99  * @param {function} callback The callback function.
100  *
101  **/
102 drupal.api.prototype.save = function(object, callback) {
103   var type = object.id ? 'PUT' : 'POST';
104   this.call(this.getURL(object), 'json', type, object, callback);
105 };
106 
107 /**
108  * API function to remove an object on the server.
109  *
110  * @param {object} object The entity object to delete.
111  * @param {function} callback The callback function.
112  */
113 drupal.api.prototype.remove = function(object, callback) {
114   this.call(this.getURL(object), 'json', 'DELETE', null, callback);
115 };
