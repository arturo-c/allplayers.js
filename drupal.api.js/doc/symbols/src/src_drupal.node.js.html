1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 /**
  5  * @constructor
  6  * @extends drupal.entity
  7  * @class The node class
  8  *
  9  * @param {object} object The node object.
 10  * @param {function} callback The function to be called once the node has
 11  * been retrieved from the server.
 12  */
 13 drupal.node = function(object, callback) {
 14 
 15   /** The title for this node. */
 16   this.title = '';
 17 
 18   /** The type of node we are dealing with. */
 19   this.type = '';
 20 
 21   /** The status of this node. */
 22   this.status = 0;
 23 
 24   /** The user who created this node */
 25   this.uid = 0;
 26 
 27   // Declare the api.
 28   this.api = new drupal.node.api();
 29 
 30   // Call the base class.
 31   drupal.entity.call(this, object, callback);
 32 };
 33 
 34 /** Derive from entity. */
 35 drupal.node.prototype = new drupal.entity();
 36 
 37 /** Reset the constructor. */
 38 drupal.node.prototype.constructor = drupal.node;
 39 
 40 /**
 41  * Override the update routine.
 42  *
 43  * @param {object} object The node object to update.
 44  */
 45 drupal.node.prototype.update = function(object) {
 46 
 47   drupal.entity.prototype.update.call(this, object);
 48 
 49   // Make sure to also set the ID the same as nid.
 50   this.id = (object && object.nid) || this.id;
 51 };
 52 
 53 /**
 54  * Returns the object to send to Services.
 55  *
 56  * @return {object} The object to send to the Services endpoint.
 57  */
 58 drupal.node.prototype.getObject = function() {
 59   return jQuery.extend(drupal.entity.prototype.getObject.call(this), {
 60     title: this.title,
 61     type: this.type,
 62     status: this.status,
 63     uid: this.uid
 64   });
 65 };
