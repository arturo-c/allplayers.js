1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 /**
  5  * @constructor
  6  * @extends drupal.entity
  7  * @class The user class
  8  *
  9  * @param {object} object The user object.
 10  * @param {function} callback The function to be called once the user has
 11  * been retrieved from the server.
 12  */
 13 drupal.user = function(object, callback) {
 14 
 15   // Only continue if the object is valid.
 16   if (object) {
 17 
 18     /** The name for this user. */
 19     this.name = this.name || '';
 20 
 21     /** The email address of our user. */
 22     this.mail = this.mail || '';
 23 
 24     /** The password of the user. */
 25     this.pass = this.pass || '';
 26 
 27     /** The status of the user. */
 28     this.status = this.status || 1;
 29 
 30     /** The session ID of the user. */
 31     this.sessid = this.sessid || '';
 32 
 33     /** The session name of the user */
 34     this.session_name = this.session_name || '';
 35 
 36     // Declare the api.
 37     this.api = this.api || new drupal.user.api();
 38   }
 39 
 40   // Call the base class.
 41   drupal.entity.call(this, object, callback);
 42 };
 43 
 44 /** Derive from drupal.entity. */
 45 drupal.user.prototype = new drupal.entity();
 46 
 47 /** Reset the constructor. */
 48 drupal.user.prototype.constructor = drupal.user;
 49 
 50 /**
 51  * Login a user.
 52  *
 53  * @param {function} callback The callback function.
 54  */
 55 drupal.user.prototype.login = function(callback) {
 56 
 57   // Setup the POST data for the login of this user.
 58   var object = {
 59     username: this.name,
 60     password: this.pass
 61   };
 62 
 63   // Execute the login.
 64   var _this = this;
 65   this.api.execute('login', object, function(user) {
 66 
 67     // Set the session ID and session name.
 68     _this.sessid = user.sessid;
 69     _this.session_name = user.session_name;
 70 
 71     // Update this object.
 72     _this.update(user.user);
 73     callback(_this);
 74   });
 75 };
 76 
 77 /**
 78  * Register a user.
 79  *
 80  * @param {function} callback The callback function.
 81  */
 82 drupal.user.prototype.register = function(callback) {
 83 
 84   // Execute the register.
 85   var _this = this;
 86   this.api.execute('register', this.getObject(), function(user) {
 87 
 88     // Now update the object.
 89     _this.update(user);
 90     callback(_this);
 91   });
 92 };
 93 
 94 /**
 95  * Logout the user.
 96  *
 97  * @param {function} callback The callback function.
 98  */
 99 drupal.user.prototype.logout = function(callback) {
100 
101   // Execute the logout.
102   this.api.execute('logout', null, callback);
103 };
104 
105 /**
106  * Override the update routine.
107  *
108  * @param {object} object The object to update.
109  */
110 drupal.user.prototype.update = function(object) {
111 
112   drupal.entity.prototype.update.call(this, object);
113 
114   // Make sure to also set the ID the same as uid.
115   if (object) {
116     this.id = object.uid || this.id;
117   }
118 };
119 
120 /**
121  * Returns the object to send to Services.
122  *
123  * @return {object} The object to send to the Services endpoint.
124  */
125 drupal.user.prototype.getObject = function() {
126   return jQuery.extend(drupal.entity.prototype.getObject.call(this), {
127     name: this.name,
128     mail: this.mail,
129     pass: this.pass,
130     status: this.status
131   });
132 };
